SELECT 
B.id,
B.name,
B.city,
B.state,
B.latitude,
B.longitude,
B.review_count,
B.stars,
B.is_open,
GROUP_CONCAT(DISTINCT(C.category)) AS categories,
GROUP_CONCAT(DISTINCT(A.name)) AS attributes
FROM business B
INNER JOIN category C
ON B.id = C.business_id
INNER JOIN attribute A
ON B.id = A.business_id
GROUP BY B.id
ORDER BY B.name;
